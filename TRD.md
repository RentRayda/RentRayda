### 2.13 Query Examples (Copy-Paste Templates)
```typescript
// packages/db/queries/listings.ts
import { db } from '../index';
import { listings, listingPhotos, landlordProfiles } from '../schema';
import { eq, and, gte, lte, desc } from 'drizzle-orm';

export async function getActiveListings(filters: {
  barangay?: string; minRent?: number; maxRent?: number;
  type?: string; page?: number;
}) {
  const page = filters.page ?? 1;
  const pageSize = 10;
  return db.query.listings.findMany({
    where: and(
      eq(listings.status, 'active'),
      filters.barangay ? eq(listings.barangay, filters.barangay) : undefined,
      filters.minRent ? gte(listings.monthlyRent, filters.minRent) : undefined,
      filters.maxRent ? lte(listings.monthlyRent, filters.maxRent) : undefined,
      filters.type ? eq(listings.unitType, filters.type) : undefined,
    ),
    with: {
      photos: { orderBy: [listingPhotos.displayOrder], limit: 1 },
      landlordProfile: {
        columns: { fullName: true, profilePhotoUrl: true, verificationStatus: true },
      },
    },
    orderBy: [desc(listings.lastActiveAt)],
    limit: pageSize,
    offset: (page - 1) * pageSize,
  });
}

// packages/db/queries/admin.ts
export async function getVerificationQueue() {
  return db.query.verificationDocuments.findMany({
    where: eq(verificationDocuments.status, 'pending'),
    with: {
      user: { columns: { phone: true, role: true } },
    },
    orderBy: [verificationDocuments.createdAt],
  });
}
```

## 3. API CONTRACTS (28 Endpoints)

Format: Method Route → Auth → Rate Limit → Zod Schema → Success → Errors

### 3.1 Auth
```
POST /api/auth/send-otp
  Auth: None | Rate: 5/15min/IP
  Zod: { phone: z.string().regex(/^09\d{9}$/) }
  200: { data: { message: "OTP sent" } }
  429: { error: "Too many OTPs. Try again later.", code: "RATE_LIMIT" }

POST /api/auth/verify-otp
  Auth: None | Rate: 10/15min/IP
  Zod: { phone: z.string().regex(/^09\d{9}$/), code: z.string().length(6),
         role: z.enum(['landlord','tenant']).optional() }
  200: { data: { user: User, isNewUser: boolean } } + Set-Cookie session
  401: { error: "Wrong code.", code: "INVALID_OTP" }
  423: { error: "Locked. Try again in 15 minutes.", code: "LOCKED_OUT" }

POST /api/auth/logout — Auth: Required → 200: { data: { message: "OK" } }
GET  /api/auth/me     — Auth: Required → 200: { data: { user, profile } }
```

### 3.2 Landlord
```
GET   /api/landlords/me       Auth: landlord → 200: LandlordProfile
PATCH /api/landlords/me       Auth: landlord → Zod: { fullName?, barangay?, unitCount?, profilePhotoUrl? }
POST  /api/landlords/verify/id       Auth: landlord → Zod: { idType, r2ObjectKey, selfieR2Key, consentAt }
  → Creates verification_documents row, sets verification_status to 'pending'
  201: { data: { documentId, status: 'pending' } }
POST  /api/landlords/verify/property Auth: landlord → Zod: { proofType, r2ObjectKey, consentAt }
  201: { data: { documentId, status: 'pending' } }
```

### 3.3 Tenant
```
GET   /api/tenants/me         Auth: tenant → 200: TenantProfile
PATCH /api/tenants/me         Auth: tenant → Zod: { fullName?, searchBarangay?, employmentType?, companyName?, profilePhotoUrl?, budgetMin?, budgetMax? }
```

```typescript
// apps/api/src/middleware/auth.ts
import { createMiddleware } from 'hono/factory';
import { auth } from '../lib/auth';

export const authMiddleware = createMiddleware(async (c, next) => {
  const session = await auth.api.getSession({ headers: c.req.raw.headers });
  if (!session) {
    return c.json({ error: 'Unauthorized', code: 'UNAUTHORIZED' }, 401);
  }
  if (session.user.isSuspended) {
    return c.json({ error: 'Account suspended', code: 'SUSPENDED' }, 403);
  }
  c.set('user', session.user);
  c.set('session', session.session);
  await next();
});

// apps/api/src/middleware/admin.ts
export const adminMiddleware = createMiddleware(async (c, next) => {
  const user = c.get('user');
  if (user.role !== 'admin') {
    return c.json({ error: 'Forbidden', code: 'NOT_ADMIN' }, 403);
  }
  await next();
});
```

```typescript
export function normalizePhPhone(raw: string): string {
  const digits = raw.replace(/[\s\-\(\)\.]/g, ''); // Strip formatting
  if (digits.startsWith('+63')) return digits;       // Already international
  if (digits.startsWith('63')) return `+${digits}`;  // Missing + prefix
  if (digits.startsWith('09')) return `+63${digits.substring(1)}`; // Local 09XX
  if (digits.startsWith('9')) return `+63${digits}`;               // No prefix
  throw new Error(`Invalid PH phone: ${raw}`);
}
```

```typescript
// apps/api/src/lib/r2.ts
import { S3Client, PutObjectCommand, GetObjectCommand, HeadObjectCommand } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';
import { randomUUID } from 'crypto';

const s3 = new S3Client({
  region: 'auto',
  endpoint: process.env.R2_ENDPOINT, // https://{account-id}.r2.cloudflarestorage.com
  credentials: {
    accessKeyId: process.env.R2_ACCESS_KEY_ID!,
    secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
  },
});

// TWO SEPARATE FUNCTIONS — never combine into one generic function
export async function generatePrivateUploadUrl(userId: string, docType: string, contentType: string) {
  const key = `${userId}/${docType}/${randomUUID()}.${contentType === 'image/png' ? 'png' : 'jpg'}`;
  const url = await getSignedUrl(s3, new PutObjectCommand({
    Bucket: process.env.R2_BUCKET_VERIFICATION!, // PRIVATE bucket
    Key: key,
    ContentType: contentType,
  }), { expiresIn: 300 }); // 5 minutes
  return { uploadUrl: url, objectKey: key };
}

export async function generatePublicUploadUrl(prefix: string, contentType: string) {
  const key = `${prefix}/${randomUUID()}.${contentType === 'image/png' ? 'png' : 'jpg'}`;
  const url = await getSignedUrl(s3, new PutObjectCommand({
    Bucket: process.env.R2_BUCKET_LISTINGS!, // PUBLIC bucket
    Key: key,
    ContentType: contentType,
  }), { expiresIn: 300 });
  return { uploadUrl: url, objectKey: key };
}

export async function generateViewUrl(key: string) {
  return getSignedUrl(s3, new GetObjectCommand({
    Bucket: process.env.R2_BUCKET_VERIFICATION!, // PRIVATE only — public has direct URLs
    Key: key,
  }), { expiresIn: 3600 }); // 1 hour for admin viewing
}

export async function verifyObjectExists(bucket: string, key: string): Promise<boolean> {
  try {
    await s3.send(new HeadObjectCommand({ Bucket: bucket, Key: key }));
    return true;
  } catch { return false; }
}
```

## 13. BUSINESS RULES (Non-Negotiable)

1. Zero payment processing — no GCash, no payment routes, no escrow

3. Government ID images in PRIVATE R2 bucket — signed URLs with 1-hour expiry

5. All `/admin/*` routes require `user.role === 'admin'`
6. Listings in search ONLY if landlord `verificationStatus === 'verified'` AND `status === 'active'`
7. Connection requests ONLY from tenants with `verificationStatus === 'verified'`
8. Landlord onboarding: max 5 screens
9. No government ID content in database — R2 keys only
10. Verification status changes ONLY via admin routes
11. Listing photos → PUBLIC bucket. Verification docs → PRIVATE bucket. Never swap.
12. New features: check this list first. If conflict, stop and get founder approval.
